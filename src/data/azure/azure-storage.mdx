Azure Storage provides two complementary services used together in Agentic AI document pipelines:

  <div className="w-1/2 min-w-0">
    <p><strong>Azure Blob Storage</strong></p>

### Generate a SAS URL

SAS (Shared Access Signature) URLs give time-limited, credential-free access to a blob — useful for passing to Document Intelligence:

<pre><code className="language-python">{`from datetime import datetime, timedelta, timezone
from azure.storage.blob import generate_blob_sas, BlobSasPermissions

def get_sas_url(blob_name: str, expiry_hours: int = 1) -> str:
    account_name = blob_service.account_name
    account_key = blob_service.credential.account_key

    sas_token = generate_blob_sas(
        account_name=account_name,
        container_name=os.environ["AZURE_STORAGE_CONTAINER"],
        blob_name=blob_name,
        account_key=account_key,
        permission=BlobSasPermissions(read=True),
        expiry=datetime.now(timezone.utc) + timedelta(hours=expiry_hours),
    )
    return f"https://{account_name}.blob.core.windows.net/{os.environ['AZURE_STORAGE_CONTAINER']}/{blob_name}?{sas_token}"`}</code></pre>

## Storage Queue

### Enqueue a Processing Task

<pre><code className="language-python">{`import json
from azure.storage.queue import QueueClient, TextBase64EncodePolicy

queue_client = QueueClient.from_connection_string(
    os.environ["AZURE_STORAGE_CONNECTION_STRING"],
    queue_name=os.environ["AZURE_STORAGE_QUEUE_NAME"],
    message_encode_policy=TextBase64EncodePolicy(),
)

# Create queue if it doesn't exist
queue_client.create_queue()

def enqueue_processing_task(blob_name: str, priority: str = "normal") -> None:
    """Tell a worker to process a newly uploaded document."""
    message = json.dumps({
        "blob_name": blob_name,
        "container": os.environ["AZURE_STORAGE_CONTAINER"],
        "priority": priority,
        "timestamp": datetime.now(timezone.utc).isoformat(),
    })
    queue_client.send_message(message)
    print(f"Queued: {blob_name}")`}</code></pre>

### Receive and Process Messages

<pre><code className="language-python">{`from azure.storage.queue import TextBase64DecodePolicy

receive_client = QueueClient.from_connection_string(
    os.environ["AZURE_STORAGE_CONNECTION_STRING"],
    queue_name=os.environ["AZURE_STORAGE_QUEUE_NAME"],
    message_decode_policy=TextBase64DecodePolicy(),
)

def process_queue(process_fn) -> None:
    """Poll queue and process each message."""
    messages = receive_client.receive_messages(max_messages=5, visibility_timeout=60)

    for msg in messages:
        try:
            task = json.loads(msg.content)
            print(f"Processing: {task['blob_name']}")
            process_fn(task)
            receive_client.delete_message(msg)
        except Exception as e:
            print(f"Error processing message: {e}")
            # Message becomes visible again after visibility_timeout
            # Consider dead-letter queue for repeated failures`}</code></pre>
