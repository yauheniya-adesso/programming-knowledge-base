Polymorphism allows objects to take many forms. The same method can behave differently based on the object calling it.

Runtime Polymorphism (Method Overriding):

class GenAIModel {
    public void generate(String prompt) {
        System.out.println("Generating output for: " + prompt);
    }
}

class TextModel extends GenAIModel {
    @Override
    public void generate(String prompt) {
        System.out.println("Generating text response for: " + prompt);
    }
}

class ImageModel extends GenAIModel {
    @Override
    public void generate(String prompt) {
        System.out.println("Generating image for: " + prompt);
    }
}

Polymorphic Behavior:

public class Main {
    public static void main(String[] args) {
        // Parent reference, child object
        GenAIModel model1 = new TextModel();
        GenAIModel model2 = new ImageModel();
        GenAIModel model3 = new GenAIModel();
        
        model1.generate("Explain polymorphism"); // Output: Generating text response for: Explain polymorphism
        model2.generate("A sunset over mountains"); // Output: Generating image for: A sunset over mountains
        model3.generate("Hello");                   // Output: Generating output for: Hello
    }
}

Polymorphism with Arrays:

public class Main {
    public static void main(String[] args) {
        GenAIModel[] models = new GenAIModel[3];
        models[0] = new TextModel();
        models[1] = new ImageModel();
        models[2] = new GenAIModel();
        
        // Polymorphic behavior in loop
        for (GenAIModel model : models) {
            model.generate("test prompt");
        }
        // Output:
        // Generating text response for: test prompt
        // Generating image for: test prompt
        // Generating output for: test prompt
    }
}

Practical Example:

class AIProvider {
    protected String modelName;
    protected double pricePerMillionTokens;
    protected int contextWindowK;
    
    public AIProvider(String modelName, double pricePerMillionTokens, int contextWindowK) {
        this.modelName = modelName;
        this.pricePerMillionTokens = pricePerMillionTokens;
        this.contextWindowK = contextWindowK;
    }
    
    public double calculateCost(int tokens) {
        return (tokens / 1_000_000.0) * pricePerMillionTokens;
    }
}

class AnthropicProvider extends AIProvider {
    private double cachedInputDiscount;
    
    public AnthropicProvider(String modelName, double price, int contextWindowK, double cachedInputDiscount) {
        super(modelName, price, contextWindowK);
        this.cachedInputDiscount = cachedInputDiscount;
    }
    
    @Override
    public double calculateCost(int tokens) {
        // Anthropic offers prompt caching discounts
        return (tokens / 1_000_000.0) * pricePerMillionTokens * (1 - cachedInputDiscount);
    }
}

class OpenAIProvider extends AIProvider {
    private int batchDiscount; // percent
    
    public OpenAIProvider(String modelName, double price, int contextWindowK, int batchDiscount) {
        super(modelName, price, contextWindowK);
        this.batchDiscount = batchDiscount;
    }
    
    @Override
    public double calculateCost(int tokens) {
        // OpenAI offers batch API discount
        return (tokens / 1_000_000.0) * pricePerMillionTokens * (1 - batchDiscount / 100.0);
    }
}

Using Polymorphism:

public class Main {
    public static void main(String[] args) {
        AIProvider[] providers = new AIProvider[3];
        providers[0] = new AIProvider("Mistral Large", 2.0, 128);
        providers[1] = new AnthropicProvider("Claude Sonnet 4.6", 3.0, 200, 0.1);
        providers[2] = new OpenAIProvider("GPT-4o", 2.5, 128, 50);
        
        int tokens = 1_000_000;
        for (AIProvider p : providers) {
            System.out.printf("%s | context: %dk | cost: $%.2f%n",
                p.modelName, p.contextWindowK, p.calculateCost(tokens));
        }
        // Output:
        // Mistral Large      | context: 128k | cost: $2.00
        // Claude Sonnet 4.6  | context: 200k | cost: $2.70
        // GPT-4o             | context: 128k | cost: $1.25
    }
}
