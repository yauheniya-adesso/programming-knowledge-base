forEach - Execute function for each element:

planets.forEach((planet, index) => {
  console.log(`${index}: ${planet}`);
});

map - Transform each element and return new array:

const planets = [
  { name: 'Earth', radius: 6371 },
  { name: 'Mars', radius: 3389 }
];
const names = planets.map(p => p.name);

filter - Return elements that pass test:

const large = planets.filter(p => p.radius > 5000);

reduce - Reduce array to single value:

const total = [1, 2, 3, 4].reduce((sum, num) => sum + num, 0);

find - Return first element that matches:

const earth = planets.find(p => p.name === 'Earth');
