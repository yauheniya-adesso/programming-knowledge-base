Objects are collections of key-value pairs and methods.

Constructor Function Pattern:

function Car(make, model, year) {
  this.make = make;
  this.model = model;
  this.year = year;
  this.getName = function() {
    return this.make + ' ' + this.model + ' ' + this.year;
  };
}

const myCar = new Car("Meridian", "Saber GT", 2012);
console.log(myCar.getName()); // "Meridian Saber GT 2012"

Object Literal:

const car = {
  make: "Toyota",
  model: "Camry",
  year: 2023,
  getName() {
    return this.make + " " + this.model + " " + this.year;
  }
};
