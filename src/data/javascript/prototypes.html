Prototypes enable inheritance in JavaScript. Objects inherit properties and methods from their prototype.

function Car(make, model, year) {
  this.make = make;
  this.model = model;
  this.year = year;
}

// Add properties/methods to prototype
Car.prototype.color = "Red";
Car.prototype.getName = function() {
  return this.make + ' ' + this.model + ' ' + this.year;
};

const car1 = new Car("Toyota", "Corolla", 2023);
console.log(car1.color); // "Red" (inherited)
console.log(car1.getName()); // "Toyota Corolla 2023"

Benefits:

  • Memory efficient (methods shared across instances)
  • Enables inheritance
  • Can be modified at runtime
