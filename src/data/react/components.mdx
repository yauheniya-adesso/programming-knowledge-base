import { useState } from 'react';

export const CounterDemo = () => {
  const [count, setCount] = useState(0);
  return (
    <div>
      <p>Count: {count}</p>
      <button className="ad-button" onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
};

export const NameInputDemo = () => {
  const [name, setName] = useState('');
  return (
    <div>
      <input
        value={name}
        onChange={e => setName(e.target.value)}
        placeholder="Type your name..."
        className="ad-input"
      />
      <p>Hello, {name || '…'}!</p>
    </div>
  );
};

React applications are built from **components** — reusable, self-contained pieces of UI. Every React app is a tree of components.

---

## Types of Components

There are three types of React components:

<div className="flex gap-4 items-start">
  <div className="w-1/3 min-w-0">
    <p><strong>Class Component</strong></p>
    <p>The original React component style (now rarely used). Extends <code>React.Component</code>, manages state via <code>this.state</code>, and requires a <code>render()</code> method.</p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`import React from 'react';

class Greeting extends React.Component {
  constructor(props) {
    super(props);
    this.state = { count: 0 };
  }

  render() {
    return (
      <div>
        <h1>Hello, {this.props.name}!</h1>
        <p>Count: {this.state.count}</p>
        <button onClick={() =>
          this.setState({ count: this.state.count + 1 })
        }>
          Increment
        </button>
      </div>
    );
  }
}

export default Greeting;`}</code></pre>
  </div>
  <div className="w-1/3 min-w-0">
    <p><strong>Function Component</strong></p>
    <p>The modern standard since React 16.8. A plain JS function (or arrow function — same behaviour, different syntax) that returns JSX and uses hooks for state and side effects.</p>
    <p><em>Standard function syntax:</em></p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useState } from 'react';

function Greeting({ name }) {
  const [count, setCount] = useState(0);

  return (
    <div>
      <h1>Hello, {name}!</h1>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>
        Increment
      </button>
    </div>
  );
}

export default Greeting;`}</code></pre>
    <p><em>Arrow function syntax (identical behaviour):</em></p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useState } from 'react';

const Greeting = ({ name }) => {
  const [count, setCount] = useState(0);

  return (
    <div>
      <h1>Hello, {name}!</h1>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>
        Increment
      </button>
    </div>
  );
};

export default Greeting;`}</code></pre>
  </div>
  <div className="w-1/3 min-w-0">
    <p><strong>Higher-Order Component (HOC)</strong></p>
    <p>A <em>pattern</em>, not a syntax. A HOC is a function that takes a component and returns a new enhanced component — used to reuse cross-cutting logic such as authentication, logging, or data fetching.</p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { Navigate } from 'react-router-dom';

// HOC: wraps any component with an auth check
function withAuth(WrappedComponent) {
  return function AuthGuard(props) {
    const isLoggedIn = !!localStorage.getItem('token');

    if (!isLoggedIn) {
      return <Navigate to="/login" replace />;
    }

    return <WrappedComponent {...props} />;
  };
}

// Usage
function Dashboard() {
  return <h1>Welcome to the dashboard!</h1>;
}

export default withAuth(Dashboard);`}</code></pre>
    <ul className="list-arrow">
      <li>HOC names conventionally start with <code>with</code></li>
      <li>The wrapped component receives all original props via <code>{`{...props}`}</code></li>
      <li>Common HOCs: <code>React.memo</code>, React Redux's <code>connect()</code></li>
    </ul>
  </div>
</div>

## Props

**Props** (short for *properties*) are read-only inputs passed from a parent component to a child. They make components reusable and configurable.

<div className="flex gap-4 items-start">
  <div className="w-1/3 min-w-0">
    <p><strong>Parent — passing props:</strong></p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`function App() {
  return (
    <UserCard
      name="Alice"
      age={30}
    />
  );
}`}</code></pre>
  </div>
  <div className="w-1/3 min-w-0">
    <p><strong>Child — receiving props:</strong></p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`function UserCard({ name, age }) {
  return (
    <div>
      <p>Name: {name}</p>
      <p>Age: {age}</p>
    </div>
  );
}`}</code></pre>
  </div>
  <div className="w-1/3 min-w-0">
    <p><strong>Rendered result:</strong></p>
    <div className="border-1 border-blue rounded-lg p-4 bg-blue-50 text-sm">
      <p>Name: Alice</p>
      <p>Age: 30</p>
    </div>
  </div>
</div>

<ul className="list-arrow">
  <li>Props flow <strong>one way</strong> — from parent to child</li>
  <li>A component must <strong>never modify</strong> its own props</li>
  <li>You can pass any value: strings, numbers, arrays, objects, or functions</li>
</ul>

## Events

React handles events using **camelCase** event handler props (e.g. `onClick`, `onChange`). Handlers are functions passed as prop values.

<div className="flex gap-4 items-start">
  <div className="w-1/2 min-w-0">
    <p><strong>Inline handler:</strong></p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`function App() {
  return (
    <button
      className="ad-button"
      onClick={() => alert('Clicked!')}
    >
      Click me
    </button>
  );
}`}</code></pre>
    <p><strong>Rendered result:</strong></p>
    <div className="border-1 border-blue rounded-lg p-4 bg-blue-50 text-sm">
      <button onClick={() => alert('Clicked!')} className="ad-button">Click me</button>
    </div>
  </div>
  <div className="w-1/2 min-w-0">
    <p><strong>Named handler:</strong></p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`function App() {
  function handleClick(e) {
    console.log(e.target);
  }

  return (
    <button className="ad-button" onClick={handleClick}>
      Click me
    </button>
  );
}`}</code></pre>
    <p><strong>Rendered result:</strong></p>
    <div className="border-1 border-blue rounded-lg p-4 bg-blue-50 text-sm">
      <button onClick={(e) => console.log(e.target)} className="ad-button">Click me</button>
      <p className="mt-2 text-brown italic">Open DevTools console to see the log</p>
    </div>
  </div>
</div>

**Common event props:**

<ul className="list-arrow">
  <li><code>onClick</code> — button / element click</li>
  <li><code>onChange</code> — input / select value change</li>
  <li><code>onSubmit</code> — form submission</li>
  <li><code>onMouseEnter</code> / <code>onMouseLeave</code> — hover</li>
  <li><code>onKeyDown</code> / <code>onKeyUp</code> — keyboard events</li>
</ul>

## State

**State** is data managed *within* a component that can change over time. When state changes, React re-renders the component.

Use the <code>useState</code> hook to add state to a function component:

<div className="flex gap-4 items-start">
  <div className="w-1/2 min-w-0">
    <p><strong>Counter example:</strong></p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Count: {count}</p>
      <button className="ad-button" onClick={() => setCount(count + 1)}>
        Increment
      </button>
    </div>
  );
}`}</code></pre>
    <p><strong>Rendered result:</strong></p>
    <div className="border-1 border-blue rounded-lg p-4 bg-blue-50 text-sm">
      <CounterDemo />
    </div>
  </div>
  <div className="w-1/2 min-w-0">
    <p><strong>Text input example:</strong></p>
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useState } from 'react';

function NameInput() {
  const [name, setName] = useState('');

  return (
    <div>
      <input
        value={name}
        onChange={e => setName(e.target.value)}
        className="ad-input"
      />
      <p>Hello, {name || '…'}!</p>
    </div>
  );
}`}</code></pre>
    <p><strong>Rendered result:</strong></p>
    <div className="border-1 border-blue rounded-lg p-4 bg-blue-50 text-sm">
      <NameInputDemo />
    </div>
  </div>
</div>

<ul className="list-arrow">
  <li><code>useState(initialValue)</code> returns <code>[currentValue, setterFunction]</code></li>
  <li>Always use the setter function — never mutate state directly</li>
  <li>State updates are <strong>asynchronous</strong> — React batches them for performance</li>
  <li>Each component instance has its <strong>own independent state</strong></li>
</ul>

## Hooks

**Hooks** are functions that let you "hook into" React features from function components. They always start with <code>use</code>.

<ul className="list-arrow">
  <li>
    <strong><code>useState</code></strong> — manage local component state (shown above)
  </li>
  <li>
    <strong><code>useEffect</code></strong> — run side effects (data fetching, subscriptions, DOM updates) after render<br />
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useEffect } from 'react';

useEffect(() => {
  document.title = 'Page loaded';
}, []); // [] = run once on mount`}</code></pre>
  </li>
  <li>
    <strong><code>useRef</code></strong> — hold a mutable value or reference a DOM element without triggering re-renders<br />
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useRef } from 'react';

const inputRef = useRef(null);
// <input ref={inputRef} />
// inputRef.current.focus();`}</code></pre>
  </li>
  <li>
    <strong><code>useContext</code></strong> — consume a React context value without prop drilling<br />
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useContext } from 'react';

const theme = useContext(ThemeContext);`}</code></pre>
  </li>
  <li>
    <strong><code>useMemo</code></strong> — memoize an expensive computed value so it only recalculates when dependencies change<br />
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useMemo } from 'react';

const sorted = useMemo(
  () => items.sort(),
  [items]
);`}</code></pre>
  </li>
  <li>
    <strong><code>useCallback</code></strong> — memoize a function reference to prevent unnecessary re-renders in child components<br />
    <pre className="overflow-x-auto"><code className="language-jsx">{`import { useCallback } from 'react';

const handleClick = useCallback(() => {
  doSomething(id);
}, [id]);`}</code></pre>
  </li>
</ul>
