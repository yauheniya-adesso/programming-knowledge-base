import { useState } from 'react';

export const CounterDemo = () => {
  const [count, setCount] = useState(0);
  return (
    <div>
      <p>Count: {count}</p>
      <button className="ad-button" onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
};

export const NameInputDemo = () => {
  const [name, setName] = useState('');
  return (
    <div>
      <input
        value={name}
        onChange={e => setName(e.target.value)}
        placeholder="Type your name..."
        className="ad-input"
      />
      <p>Hello, {name || '…'}!</p>
    </div>
  );
};

**State** is data managed *within* a component that can change over time. When state changes, React re-renders the component.

Use the <code>useState</code> hook to add state to a function component:

<ul className="list-arrow">
  <li><code>useState(initialValue)</code> returns <code>[currentValue, setterFunction]</code></li>
  <li>Always use the setter function — never mutate state directly</li>
  <li>State updates are <strong>asynchronous</strong> — React batches them for performance</li>
  <li>Each component instance has its <strong>own independent state</strong></li>
</ul>
